\section{Data Sources}\label{app.data}
\newcommand{\q}[3]{#1 [#2] \shortquote{#3}}
The lists below give the exact wording of survey questions used in our analysis.
Additional information on survey design, implementation, and results
are available in the respective publications
\cite{Baral2014} (2011) and \cite{EswKP2014} (2014).
\paragraph{Duration Selling Sex}
\begin{itemize}
  \item {current age}
  \begin{itemize}
    \item \q{2011}{1.1}{How old are you?}
    \item \q{2014}{8}{How old are you?}
  \end{itemize}
  \item {age first sold sex}
  \begin{itemize}
    \item \q{2011}{3.00}{How old were you the first time you traded or sold sex or sexual acts in exchange for money, favours or goods?}
    \item \q{2014}{4.1}{Approximately how old were you the first time you provided sexual acts in exchange for money?}
  \end{itemize}
\end{itemize}
\paragraph{Numbers of Sexual Partners}
\begin{itemize}
  \item {new clients}
  \begin{itemize}
    \item \q{2011}{4.06a}{In the past 30 days, how many new clients have you had?}
    \item \q{2014}{5.5}{In the last month, approximately how many new clients did you have any type of sex with (vaginal, anal, oral)?}
  \end{itemize}
  \item {regular clients}
  \begin{itemize}
    \item \q{2011}{4.11a}{In the last 30 days, how many regular clients have you had sex with?}
    \item \q{2014}{5.15}{In the last 30 days, approximately how many regular clients have you had any type of sex with (including oral, vaginal, or anal sex)?}
  \end{itemize}
  \item {non-paying partners}
  \begin{itemize}
    \item \q{2011}{4.17a}{In the last 30 days, how many non-paying partners have you had sex with?}
    \item \q{2014}{5.25}{In the last month, approximately how many non-paying partners have you had sex with?}
  \end{itemize}
\end{itemize}
